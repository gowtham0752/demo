fun getLoaderInfo() {
        if (config.icon.isNotEmpty() && config.msg.isNotEmpty() && config.colorCode.isNotEmpty()) {
            val imageBytes = Base64.decode(config.icon, Base64.DEFAULT)
            val bitmap = BitmapFactory.decodeByteArray(imageBytes, 0, imageBytes.size)
            img.setImageBitmap(bitmap)
            linear_color.setBackgroundColor(Color.parseColor(config.colorCode))
            txt.text = config.msg
            loading.visibility = VISIBLE
        } else if (config.icon.isNotEmpty() && config.msg.isNotEmpty() && config.colorCode.isEmpty()) {
            val imageBytes = Base64.decode(config.icon, Base64.DEFAULT)
            val bitmap = BitmapFactory.decodeByteArray(imageBytes, 0, imageBytes.size)
            img.setImageBitmap(bitmap)
            linear_color.setBackgroundColor(Color.parseColor(DEFAULT_LOADER_COLOR))
            txt.text = config.msg
            loading.visibility = VISIBLE
        } else if (config.icon.isNotEmpty() && config.msg.isEmpty() && config.colorCode.isNotEmpty()) {
            val imageBytes = Base64.decode(config.icon, Base64.DEFAULT)
            val bitmap = BitmapFactory.decodeByteArray(imageBytes, 0, imageBytes.size)
            img.setImageBitmap(bitmap)
            linear_color.setBackgroundColor(Color.parseColor(config.colorCode))
            txt.visibility = GONE
            loading.visibility = VISIBLE
        } else if (config.icon.isEmpty() && config.msg.isNotEmpty() && config.colorCode.isNotEmpty()) {
            img.visibility = GONE
            linear_color.setBackgroundColor(Color.parseColor(config.colorCode))
            txt.text = config.msg
            loading.visibility = VISIBLE
        } else if (config.icon.isNotEmpty() && config.msg.isEmpty() && config.colorCode.isEmpty()) {
            val imageBytes = Base64.decode(config.icon, Base64.DEFAULT)
            val bitmap = BitmapFactory.decodeByteArray(imageBytes, 0, imageBytes.size)
            img.setImageBitmap(bitmap)
            linear_color.setBackgroundColor(Color.parseColor(DEFAULT_LOADER_COLOR))
            txt.visibility = GONE
            loading.visibility = VISIBLE
        } else if ( config.icon.isEmpty() && config.msg.isNotEmpty() &&  config.colorCode.isEmpty()) {
            img.visibility = GONE
            linear_color.setBackgroundColor(Color.parseColor(DEFAULT_LOADER_COLOR))
            txt.text = config.msg
            loading.visibility = VISIBLE
        } else {
            loading.visibility = GONE
            progress_loading.visibility = VISIBLE
        }
    }
